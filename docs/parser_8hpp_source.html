19 template <typename Scalar>
20 Eigen::Matrix<Scalar, 3, 1> vec_from_string(const std::string& vector_str) {
21 Eigen::Matrix<Scalar, 3, 1> vec;
22 std::vector<Scalar> values;
23 std::istringstream ss(vector_str);
24 std::string vector_element;
25 while (ss >> vector_element) {
26 try {
27 values.push_back(std::stod(vector_element));
28 }
29 catch (std::invalid_argument& e) {
30 throw std::runtime_error("Error not able to parse component (" + vector_element
31 + ") to a Scalar (while parsing a vector value)");
32 }
33 }
34
35 if (values.size() != 3) {
36 std::ostringstream error_msg;
37 error_msg << "Parser found " << values.size() << " elements but 3 expected while parsing vector ["
38 << vector_str << "]";
39 throw std::runtime_error(error_msg.str());
40 }
41
42 vec.x() = values[0];
43 vec.y() = values[1];
44 vec.z() = values[2];
45
46 return vec;
47 }

55 Eigen::Matrix<Scalar, 3, 3> rot_from_string(const std::string& rotation_str) {
56 Eigen::Matrix<Scalar, Eigen::Dynamic, 1> rpy = vec_from_string<Scalar>(rotation_str);
57 Eigen::Matrix<Scalar, 3, 3> R =
58 Eigen::AngleAxis<Scalar>(rpy.z(), Eigen::Matrix<Scalar, 3, 1>::UnitZ())
59 * Eigen::AngleAxis<Scalar>(rpy.y(), Eigen::Matrix<Scalar, 3, 1>::UnitY())
60 * Eigen::AngleAxis<Scalar>(rpy.x(), Eigen::Matrix<Scalar, 3, 1>::UnitX()).toRotationMatrix();
61 return R;
62 }

93 Eigen::Matrix<Scalar, 6, 6> spatial_transform_from_xml(tinyxml2::XMLElement* xml) {
94 Eigen::Matrix<Scalar, 6, 6> X = Eigen::Matrix<Scalar, 6, 6>::Identity();
95 Eigen::Matrix<Scalar, 6, 6> R = Eigen::Matrix<Scalar, 6, 6>::Identity();
96 Eigen::Matrix<Scalar, 6, 6> T = Eigen::Matrix<Scalar, 6, 6>::Identity();
97 if (xml) {
98 const char* rpy_str = xml->Attribute("rpy");
99 if (rpy_str != nullptr) {

101 R = rpy_to_spatial(rpy);
102 }
103
104 const char* xyz_str = xml->Attribute("xyz");
105 if (xyz_str != nullptr) {
106 Eigen::Matrix<Scalar, 3, 1> translation = vec_from_string<Scalar>(xyz_str);
107 T = translation_to_spatial(translation);
108 }
109 }
110 X = R * T;

136 Link<Scalar> link_from_xml(tinyxml2::XMLElement* xml) {
137 // Create the link object
138 Link<Scalar> link = Link<Scalar>();
139
140 const char* name_char = xml->Attribute("name");
141 if (name_char != nullptr) {
142 link.name = std::string(name_char);
143 }
144 else {
145 std::ostringstream error_msg;
146 error_msg << "Error! Link without a name attribute detected!";
147 throw std::runtime_error(error_msg.str());
148 }
149
150 tinyxml2::XMLElement* i = xml->FirstChildElement("inertial");
151 if (i != nullptr) {
152 // Add the centre of mass to the link
153 tinyxml2::XMLElement* o = i->FirstChildElement("origin");
154 if (o != nullptr) {
155 link.centre_of_mass = transform_from_xml<Scalar>(o);
156 }
157 // Add the mass to the link
158 tinyxml2::XMLElement* mass_xml = i->FirstChildElement("mass");
159 if (mass_xml != nullptr) {
160 if (mass_xml->Attribute("value") != nullptr) {
161 try {
162 link.mass = std::stod(mass_xml->Attribute("value"));
163 }
164 catch (std::invalid_argument& e) {
165 std::ostringstream error_msg;
166 error_msg << "Error while parsing link '" << get_parent_link_name(i) << "': inertial mass ["
167 << mass_xml->Attribute("value") << "] is not a valid double: " << e.what() << "!";
168 throw std::runtime_error(error_msg.str());
169 }
170 }
171 else {
172 std::ostringstream error_msg;
173 error_msg << "Error while parsing link '" << get_parent_link_name(i)
174 << "' <mass> element must have a value attribute!";
175 throw std::runtime_error(error_msg.str());
176 }
177 }
178 else {
179 std::ostringstream error_msg;
180 error_msg << "Error while parsing link '" << get_parent_link_name(i)
181 << "' inertial element must have a <mass> element!";
182 throw std::runtime_error(error_msg.str());
183 }
184 }
185 // Add the inertia to the link
186 tinyxml2::XMLElement* inertia_xml = i->FirstChildElement("inertia");
187 if (inertia_xml != nullptr) {
188 Eigen::Matrix<Scalar, 3, 3> inertia;
189 if (inertia_xml->Attribute("ixx") && inertia_xml->Attribute("ixy") && inertia_xml->Attribute("ixz")
190 && inertia_xml->Attribute("iyy") && inertia_xml->Attribute("iyz") && inertia_xml->Attribute("izz")) {
191 try {
192 inertia(0, 0) = std::stod(inertia_xml->Attribute("ixx"));
193 inertia(0, 1) = std::stod(inertia_xml->Attribute("ixy"));
194 inertia(0, 2) = std::stod(inertia_xml->Attribute("ixz"));
195 inertia(1, 0) = std::stod(inertia_xml->Attribute("ixy"));
196 inertia(1, 1) = std::stod(inertia_xml->Attribute("iyy"));
197 inertia(1, 2) = std::stod(inertia_xml->Attribute("iyz"));
198 inertia(2, 0) = std::stod(inertia_xml->Attribute("ixz"));
199 inertia(2, 1) = std::stod(inertia_xml->Attribute("iyz"));
200 inertia(2, 2) = std::stod(inertia_xml->Attribute("izz"));
201 link.inertia = inertia;
202 }
203 catch (std::invalid_argument& e) {
204 std::ostringstream error_msg;
205 error_msg << "Error while parsing link '" << get_parent_link_name(i)
206 << "Inertial: one of the inertia elements is not a valid double:"
207 << " ixx [" << inertia_xml->Attribute("ixx") << "]"
208 << " ixy [" << inertia_xml->Attribute("ixy") << "]"
209 << " ixz [" << inertia_xml->Attribute("ixz") << "]"
210 << " iyy [" << inertia_xml->Attribute("iyy") << "]"
211 << " iyz [" << inertia_xml->Attribute("iyz") << "]"
212 << " izz [" << inertia_xml->Attribute("izz") << "]\n\n"
213 << e.what();
214 throw std::runtime_error(error_msg.str());
215 }
216 }
217 else {
218 std::ostringstream error_msg;
219 error_msg << "Error while parsing link '" << get_parent_link_name(i)
220 << "' <inertia> element must have ixx,ixy,ixz,iyy,iyz,izz attributes!";
221 throw std::runtime_error(error_msg.str());
222 }
223 }
224 else {
225 std::ostringstream error_msg;
226 error_msg << "Error while parsing link '" << get_parent_link_name(i)
227 << "' inertial element must have a <inertia> element!";
228 throw std::runtime_error(error_msg.str());
229 }
230
231 // Add the spatial inertia to the link
232 link.I = inertia_to_spatial<Scalar>(link.mass, link.centre_of_mass.translation(), link.inertia);
233
234 return link;
235 }

365 template <typename Scalar, int nq>
366 void init_link_tree(Model<Scalar, nq>& model) {
367 // Initialize the joint count to zero
368 model.n_q = 0;
369
370 // Iterate over each joint in the model
371 for (auto joint : model.joints) {
372 // Check that the joint has a parent link and a child link specified
373 std::string parent_link_name = joint.parent_link_name;
374 if (parent_link_name.empty()) {
375 std::ostringstream error_msg;
376 error_msg << "Error while constructing model! Joint [" << joint.name
377 << "] is missing a parent link specification.";
378 throw std::runtime_error(error_msg.str());
379 }
380 std::string child_link_name = joint.child_link_name;
381 if (child_link_name.empty()) {
382 std::ostringstream error_msg;
383 error_msg << "Error while constructing model! Joint [" << joint.name
384 << "] is missing a child link specification.";
385 throw std::runtime_error(error_msg.str());
386 }
387
388 // Get references to the child and parent links associated with the joint
389 auto child_link = model.get_link(child_link_name);
390 if (child_link.idx == -1) {
391 std::ostringstream error_msg;
392 error_msg << "Error while constructing model! Child link [" << child_link_name << "] of joint ["
393 << joint.name << "] not found";
394 throw std::runtime_error(error_msg.str());
395 }
396
397 auto parent_link = model.get_link(parent_link_name);
398 if (parent_link.idx == -1) {
399 std::ostringstream error_msg;
400 error_msg << "Error while constructing model! Parent link [" << parent_link_name << "] of joint ["
401 << joint.name << "] not found";
402 throw std::runtime_error(error_msg.str());
403 }
404
405 // Set the parent link index for the child link and add the child link index to the parent link's list
406 // of child link indices
407 child_link.parent = parent_link.idx;
408 parent_link.add_child_link_idx(child_link.idx);
409
410 // If the joint is of type REVOLUTE or PRISMATIC, add its index in the configuration vector and
411 // increment the configuration vector size
412 if (joint.type == JointType::REVOLUTE || joint.type == JointType::PRISMATIC) {
413 joint.idx = model.n_q;
414 model.n_q++;
415 }
416
417 // Associate the joint with the child link and update the child link in the links vector
418 child_link.joint = joint;
419 model.links[child_link.idx] = child_link;
420
421 // Update the parent link in the links vector
422 model.links[parent_link.idx] = parent_link;
423 }
424
425 // Find the base link of the model by finding the link with no parent link
426 for (auto link : model.links) {
427 bool found = false;
428 if (link.parent == -1) {
429 if (found) {
430 throw std::runtime_error(
431 "Error! Multiple base links found. The urdf does not contain a valid link tree.");
432 }
433 model.base_link_idx = link.idx;
434 found = true;
435 }
436 }

446 template <typename Scalar, int nq>
447 void init_dynamics(Model<Scalar, nq>& model) {
448 for (auto& link : model.links) {
449 // If the link has a fixed joint, update the transforms of the child links and its parents link inertia
450 if (link.joint.type == JointType::FIXED && link.idx != model.base_link_idx) {
451 // Update fixed transforms of the child links
452 for (auto child_link_idx : link.child_links) {
453 auto child_link = model.links[child_link_idx];
454 for (int j = 0; j < model.links.size(); j++) {
455 if (model.links[j].name == child_link.name) {
456 // TODO: IMPLEMENT SOMETHING LIKE THIS
457 model.links[j].joint.X = model.links[j].joint.X * link.joint.X;
458 break;
459 }
460 }
461 }
462 // Combine spatial inertias
463 auto parent_link = model.links[link.parent];
464 // Add the spatial inertia of the link to its parent link in the link tree
465 for (int j = 0; j < model.links.size(); j++) {
466 if (model.links[j].name == parent_link.name) {
467 Eigen::Matrix<Scalar, 6, 6> X_T = model.links[j].joint.X;
468 model.links[j].I += X_T.transpose() * link.I * X_T;
469 break;
470 }
471 }
472 }
473 else if (link.idx != model.base_link_idx) {
474 // Add the index of this actuatable link to the list of dynamic link indices
475 model.q_idx.push_back(link.idx);
476 }
477 }
478 // For each dynamic link, find the index of its parent link in the dynamic link indices
479 for (int i = 0; i < model.q_idx.size(); i++) {
480 // Get the dynamic link
481 auto link = model.links[model.q_idx[i]];
482 // Get the parent link which isn't fixed
483 auto parent_link = model.links[link.parent];
484 while (parent_link.joint.type == JointType::FIXED) {
485 parent_link = model.links[parent_link.parent];
486 }
487 // First check to see if the parent link is the base link
488 if (parent_link.idx == model.base_link_idx) {
489 // Add the index of the base link to the list of parent indices
490 model.parent.push_back(-1);
491 }
492 else {
493 // Find the parent link in the q_idx
494 bool found = false;
495 for (int j = 0; j < model.q_idx.size(); j++) {
496 if (model.q_idx[j] == parent_link.idx) {
497 model.parent.push_back(j);
498 found = true;
499 break;
500 }
501 }
502 if (!found) {
503 model.parent.push_back(-1);
504 }
505 }
506 }
507
508 // Assign spatial gravity vector in models data
509 model.data.spatial_gravity.tail(3) = model.gravity;
510
511 // Compute the total mass of the model
512 model.mass = 0;
513 for (auto link : model.links) {
514 model.mass += link.mass;
515 }
516 }

523 template <typename Scalar, int nq>
524 Model<Scalar, nq> import_urdf(const std::string& path_to_urdf) {
525 // Open the URDF file
526 std::ifstream input_file(path_to_urdf);
527 if (!input_file.is_open()) {
528 throw std::runtime_error("Could not open the file - '" + path_to_urdf + "'");
529 }
530
531 // Read the URDF file into an XML string
532 std::string urdf_string =
533 std::string(std::istreambuf_iterator<char>(input_file), std::istreambuf_iterator<char>());
534
535 // Parse the XML string using tinyxml2
536 tinyxml2::XMLDocument xml_doc;
537 xml_doc.Parse(urdf_string.c_str());
538 if (xml_doc.Error()) {
539 throw std::runtime_error("Error parsing XML: " + std::string(xml_doc.ErrorStr()));
540 }
541
542 // Get the robot element
543 tinyxml2::XMLElement* robot_xml = xml_doc.RootElement();
544 if (!robot_xml || std::string(robot_xml->Value()) != "robot") {
545 throw std::runtime_error("Error: Could not find the <robot> element in the URDF file");
546 }
547
548 // Create a new model
549 Model<Scalar, nq> model;
550
551 // Set the models name
552 const char* name = robot_xml->Attribute("name");
553 if (!name) {
554 throw std::runtime_error("Error: No name given for the robot");
555 }
556 model.name = name;
557
558 // Parse the links
559 for (tinyxml2::XMLElement* link_xml = robot_xml->FirstChildElement("link"); link_xml;
560 link_xml = link_xml->NextSiblingElement("link")) {
561 Link<Scalar> link = link_from_xml<Scalar>(link_xml);
562 if (model.get_link(link.name).idx != -1) {
563 throw std::runtime_error("Error: Duplicate links '" + link.name + "' found");
564 }
565 link.idx = model.links.size();
566 model.links.push_back(link);
567 }
568 if (model.links.empty()) {
569 throw std::runtime_error("Error: No link elements found in the URDF file");
570 }
571
572 // Parse the joints
573 for (tinyxml2::XMLElement* joint_xml = robot_xml->FirstChildElement("joint"); joint_xml;
574 joint_xml = joint_xml->NextSiblingElement("joint")) {
575 Joint<Scalar> joint = joint_from_xml<Scalar>(joint_xml);
576 if (model.get_joint(joint.name).joint_id != -1) {
577 throw std::runtime_error("Error: Duplicate joints '" + joint.name + "' found");
578 }
579 joint.joint_id = model.joints.size();
580 model.joints.push_back(joint);
581 }
582
583 // Initialize the link tree and find the base link index (should be -1)
584 init_link_tree(model);
585
586 // Initialize the q_idx and parent_map
587 init_dynamics(model);
588
589 // Set the number of joints and links
590 model.n_joints = model.joints.size();
591 model.n_links = model.links.size();
592
593 return model;
594 }
