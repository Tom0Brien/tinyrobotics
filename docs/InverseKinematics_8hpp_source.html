167 static inline Eigen::Matrix<AutoDiffType, 1, 1> cost(
168 const Eigen::Matrix<AutoDiffType, nq, 1>& q,
169 const Model<AutoDiffType, nq>& model,
170 const std::string& source_link_name,
171 const std::string& target_link_name,
172 const Eigen::Transform<Scalar, 3, Eigen::Isometry> Hst_desired,
173 const Eigen::Matrix<AutoDiffType, nq, 1>& q0) {
174 // Compute the forward kinematics from the source link to the target link using the current joint angles.
175 Eigen::Transform<AutoDiffType, 3, Eigen::Isometry> Hst_current =
176 forward_kinematics<AutoDiffType, nq>(model, q, source_link_name, target_link_name);
177
178 // Compute the euler angles for current
179 Eigen::Matrix<AutoDiffType, 3, 3> Rst_current = Hst_current.linear();
180 // Compute the euler angles for desired
181 Eigen::Matrix<AutoDiffType, 3, 3> Rst_desired = Hst_desired.linear();
182 Eigen::Matrix<AutoDiffType, 3, 3> R_v_r = Rst_desired * Rst_current.transpose();
183
184 Eigen::Matrix<AutoDiffType, 1, 1> o_error = Eigen::Matrix<AutoDiffType, 1, 1>(
185 (Eigen::Matrix<AutoDiffType, 3, 3>::Identity() - R_v_r).diagonal().sum());
186
187 Eigen::Matrix<AutoDiffType, nq, 1> q_diff = q - q0;
188
189 // Quadratic cost function q^T*W*q + (k(q) - x*)^TK*(l(q) - x*)))
190 Eigen::Matrix<AutoDiffType, nq, nq> W = Eigen::Matrix<AutoDiffType, nq, nq>::Identity(model.n_q, model.n_q);
191 Eigen::Matrix<AutoDiffType, 3, 3> K = Eigen::Matrix<AutoDiffType, 3, 3>::Identity();
192
193 // Compute the cost function
194 Eigen::Matrix<AutoDiffType, 1, 1> cost =
195 ((Hst_current.translation() - Hst_desired.translation()).transpose() * 1e2 * K
196 * (Hst_current.translation() - Hst_desired.translation()))
197 + (q - q0).transpose() * 1e-1 * W * (q - q0) + o_error.transpose() * 1e3 * o_error;
198
199 return cost;
200 }

221 void FillJacobianBlock(std::string var_set, Jacobian& jac) const override {
222 if (var_set == "configuration_vector") {
223 // Cast q and model to autodiff type
224 Eigen::Matrix<AutoDiffType, nq, 1> q_auto(
225 GetVariables()->GetComponent("configuration_vector")->GetValues());
226
227 // The output vector F = f(x) evaluated together with Jacobian matrix below
228 Eigen::Matrix<AutoDiffType, 1, 1> F;
229
230 // Compute the Jacobian
231 Eigen::Matrix<Scalar, 1, nq> J =
232 jacobian(cost,
233 wrt(q_auto),
234 at(q_auto, model, source_link_name, target_link_name, desired_pose, q0),
235 F);
236
237 // Fill the Jacobian block
238 jac.resize(J.rows(), J.cols());

256 Eigen::Matrix<Scalar, nq, 1> inverse_kinematics(Model<Scalar, nq>& model,
257 std::string& source_link_name,
258 std::string& target_link_name,
259 const Eigen::Transform<Scalar, 3, Eigen::Isometry>& desired_pose,
260 Eigen::Matrix<Scalar, nq, 1> q0) {
261
262 // Cast model to autodiff type

264 // 1. Define the problem
265 ifopt::Problem nlp;
266 nlp.AddVariableSet(std::make_shared<IKVariables<double, nq>>("configuration_vector", model, q0));
267 // nlp.AddConstraintSet(std::make_shared<IKConstraint<double>>());
268 nlp.AddCostSet(std::make_shared<IKCost<double, autodiff::dual, nq>>("IK_cost",
269 autodiff_model,
270 source_link_name,
271 target_link_name,
272 desired_pose,
273 q0));
274 // 2. Choose solver and options
275 ifopt::IpoptSolver ipopt;
276 ipopt.SetOption("mu_strategy", "adaptive");
277 ipopt.SetOption("jacobian_approximation", "exact");
278 ipopt.SetOption("max_iter", 1000);
279 ipopt.SetOption("tol", 1e-4);
280 ipopt.SetOption("print_level", 0);
281 ipopt.SetOption("sb", "yes");
282 // 3. Solve
283 ipopt.Solve(nlp);
284 Eigen::Matrix<Scalar, nq, 1> q = nlp.GetOptVariables()->GetValues();
